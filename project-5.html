#include <WiFi.h>
#include "AdafruitIO_WiFi.h"
#include <Adafruit_Fingerprint.h>

// Adafruit IO credentials
#define IO_USERNAME "IoTTrainingBoard"
#define IO_KEY "aio_Ceey02l0UmBmqgSj95VjH1090iRC"

// WiFi credentials
#define WIFI_SSID "PLDTHOMEFIBR5e300"
#define WIFI_PASS "SASHuke@1123"

// Adafruit IO instance
AdafruitIO_WiFi io(IO_USERNAME, IO_KEY, WIFI_SSID, WIFI_PASS);

// Feeds for library management
AdafruitIO_Feed *attendanceFeed = io.feed("attendance_logs");
AdafruitIO_Feed *statusFeed = io.feed("system_status");
AdafruitIO_Feed *registerUserFeed = io.feed("register_user");
AdafruitIO_Feed *bookEntryFeed = io.feed("book_entry");
AdafruitIO_Feed *availableBooksFeed = io.feed("available_books");

// Fingerprint Sensor Configuration
#define RX_PIN 16
#define TX_PIN 17
Adafruit_Fingerprint finger = Adafruit_Fingerprint(&Serial2);

String currentMessage = "Waiting for fingerprint...";
String logEntries = "";
bool isRegistering = false;
uint8_t userID = 1;

// Arrays for users and books
const int MAX_USERS = 20;
uint8_t userIDs[MAX_USERS];
String userNames[MAX_USERS];
int registeredUserCount = 0;

const int MAX_BOOKS = 10;
String books[MAX_BOOKS] = {"Book A", "Book B", "Book C", "Book D", "Book E"};
bool bookAvailable[MAX_BOOKS] = {true, true, true, true, true};

// Add a user
void addUser(uint8_t id, String name) {
  if (registeredUserCount < MAX_USERS) {
    userIDs[registeredUserCount] = id;
    userNames[registeredUserCount] = name;
    registeredUserCount++;
    Serial.println("User added: " + name);
  }
}

// Get user name by ID
String getUserNameByID(uint8_t id) {
  for (int i = 0; i < registeredUserCount; i++) {
    if (userIDs[i] == id) {
      return userNames[i];
    }
  }
  return "Unknown User";
}

// Update the list of available books
void updateAvailableBooks() {
  String bookList = "";
  for (int i = 0; i < MAX_BOOKS; i++) {
    if (bookAvailable[i]) {
      bookList += books[i] + "\n";
    }
  }
  if (bookList == "") {
    bookList = "No books available.";
  }
  availableBooksFeed->save(bookList); // Update Adafruit IO feed
}

// Register fingerprint
void registerFingerprint(String userName) {
  isRegistering = true;
  currentMessage = "Place your finger to register...";
  statusFeed->save(currentMessage);

  if (finger.getImage() != FINGERPRINT_OK) {
    currentMessage = "Failed to capture fingerprint.";
    statusFeed->save(currentMessage);
    return;
  }

  if (finger.image2Tz() != FINGERPRINT_OK) {
    currentMessage = "Failed to process fingerprint.";
    statusFeed->save(currentMessage);
    return;
  }

  if (finger.storeModel(userID) == FINGERPRINT_OK) {
    addUser(userID, userName);
    currentMessage = "Registered successfully: " + userName;
    attendanceFeed->save(userName + " registered.");
    userID++;
  } else {
    currentMessage = "Failed to store fingerprint.";
  }

  statusFeed->save(currentMessage);
  isRegistering = false;
}

// Check fingerprint
void checkFingerprint() {
  if (isRegistering) return;

  if (finger.getImage() != FINGERPRINT_OK) return;

  if (finger.image2Tz() != FINGERPRINT_OK) return;

  int p = finger.fingerSearch();
  if (p == FINGERPRINT_OK) {
    uint8_t id = finger.fingerID;
    String userName = getUserNameByID(id);

    currentMessage = userName + " recognized. Enter a book title to borrow/return.";
    statusFeed->save(currentMessage);
  } else {
    currentMessage = "Fingerprint not recognized.";
    statusFeed->save(currentMessage);
  }
}

// Borrow or return a book
void handleBookSubmission(String bookTitle) {
  bool found = false;
  for (int i = 0; i < MAX_BOOKS; i++) {
    if (books[i] == bookTitle) {
      found = true;
      if (bookAvailable[i]) {
        bookAvailable[i] = false;
        logEntries += bookTitle + " borrowed<br>";
        attendanceFeed->save(bookTitle + " borrowed.");
        currentMessage = bookTitle + " borrowed.";
      } else {
        bookAvailable[i] = true;
        logEntries += bookTitle + " returned<br>";
        attendanceFeed->save(bookTitle + " returned.");
        currentMessage = bookTitle + " returned.";
      }
      break;
    }
  }

  if (!found) {
    currentMessage = "Book not found.";
  }

  updateAvailableBooks(); // Update the list of available books
  statusFeed->save(currentMessage);
}

// Handle book entry via Adafruit IO
void handleBookEntry(AdafruitIO_Data *data) {
  String bookTitle = data->toString();
  handleBookSubmission(bookTitle);
}

// Handle Adafruit IO registration messages
void handleRegistration(AdafruitIO_Data *data) {
  String userName = data->toString();
  registerFingerprint(userName);
}

void setup() {
  Serial.begin(115200);

  // Initialize fingerprint sensor
  Serial2.begin(57600, SERIAL_8N1, RX_PIN, TX_PIN);
  finger.begin(57600);
  if (finger.verifyPassword()) {
    Serial.println("Fingerprint sensor ready.");
  } else {
    Serial.println("Fingerprint sensor not detected.");
    while (1);
  }

  // Connect to Adafruit IO
  io.connect();
  registerUserFeed->onMessage(handleRegistration);
  bookEntryFeed->onMessage(handleBookEntry);

  while (io.status() < AIO_CONNECTED) {
    delay(500);
  }
  Serial.println("Connected to Adafruit IO.");

  // Initialize feeds
  attendanceFeed->save("System started.");
  statusFeed->save(currentMessage);
  updateAvailableBooks(); // Initialize the list of available books
}

void loop() {
  io.run();
  checkFingerprint();
}

// Define the Access Point credentials

String logEntries = ""; // To store book check-out/in logs
bool isRegistering = false;  // Flag to indicate registration mode
bool nameEntryRequired = false; // Flag to indicate if name entry is needed
bool bookEntryRequired = false; // Flag to indicate if book entry is needed
uint8_t userID = 1; // ID to assign to new registered fingerprints
int registrationStage = 0; // Track fingerprint registration stages

String userNames[MAX_USERS];

// Function to get the list of available books

// Function to initiate registration
void initiateRegistration() {
  isRegistering = true;
  registrationStage = 0;
  currentMessage = "Registration started. Place your finger.";
}

// Function to cancel registration
void cancelRegistration() {
  isRegistering = false;
  nameEntryRequired = false;
  bookEntryRequired = false;
  registrationStage = 0;
  currentMessage = "Registration canceled. Waiting for fingerprint...";
}

// Function to handle fingerprint registration process
void registerFingerprint() {
  if (!isRegistering) return;

  int p;
  switch (registrationStage) {
    case 0: // Get image stage
      currentMessage = "Place your finger to register...";
      p = finger.getImage();
      if (p == FINGERPRINT_OK) {
        registrationStage++;
        currentMessage = "Fingerprint image taken. Remove finger.";
      } else if (p == FINGERPRINT_NOFINGER) {
        // Do nothing, waiting for the finger
      } else {
        currentMessage = "Failed to read fingerprint. Try again.";
        isRegistering = false;
        registrationStage = 0;
      }
      break;
    
    case 1: // Convert image to template
      p = finger.image2Tz();
      if (p == FINGERPRINT_OK) {
        registrationStage++;
        currentMessage = "Fingerprint converted. Place the same finger again.";
      } else {
        currentMessage = "Failed to convert fingerprint. Try again.";
        isRegistering = false;
        registrationStage = 0;
      }
      break;
    
    case 2: // Get second image
      p = finger.getImage();
      if (p == FINGERPRINT_OK) {
        registrationStage++;
        currentMessage = "Second fingerprint image taken. Remove finger.";
      } else if (p == FINGERPRINT_NOFINGER) {
        // Do nothing, waiting for the finger
      } else {
        currentMessage = "Failed to read fingerprint. Try again.";
        isRegistering = false;
        registrationStage = 0;
      }
      break;
    
    case 3: // Convert second image to template
      p = finger.image2Tz(2);
      if (p == FINGERPRINT_OK) {
        registrationStage++;
        currentMessage = "Second fingerprint converted. Creating model.";
      } else {
        currentMessage = "Failed to convert second fingerprint. Try again.";
        isRegistering = false;
        registrationStage = 0;
      }
      break;
    
    case 4: // Create a model
      p = finger.createModel();
      if (p == FINGERPRINT_OK) {
        registrationStage++;
        currentMessage = "Model created. Storing fingerprint.";
      } else {
        currentMessage = "Failed to create fingerprint model. Try again.";
        isRegistering = false;
        registrationStage = 0;
      }
      break;
    
    case 5: // Store the model
      p = finger.storeModel(userID);
      if (p == FINGERPRINT_OK) {
        currentMessage = "Fingerprint registered successfully! Please enter your name.";
        nameEntryRequired = true; // Enable name entry
        isRegistering = false; // End fingerprint registration
        registrationStage = 0; // Reset the registration stage
      } else {
        currentMessage = "Failed to store fingerprint. Try again.";
        isRegistering = false;
        registrationStage = 0;
      }
      break;
  }
}

// Function to check fingerprint and handle borrow/return logic
void checkFingerprint() {
  if (isRegistering || nameEntryRequired || bookEntryRequired) {
    return; // Skip detection if in registration or name/book entry mode
  }

  if (finger.getImage() != FINGERPRINT_OK) return;
  if (finger.image2Tz() != FINGERPRINT_OK) return;

  int p = finger.fingerSearch();
  if (p == FINGERPRINT_OK) {
    uint8_t id = finger.fingerID;
    String userName = getUserNameByID(id);

    // Enable the name and book input forms
    currentMessage = "Welcome, " + userName + "! Enter the book title to borrow or return.";
    bookEntryRequired = true;
  } else {
    currentMessage = "Fingerprint not recognized. Do you want to register?";
    isRegistering = false;
  }
}

// Handle name submission
void handleNameSubmission() {
  if (!nameEntryRequired) return;

  if (server.hasArg("name")) {
    String userName = server.arg("name");
    addUser(userID, userName);
    currentMessage = "User " + userName + " registered successfully! Enter the book title.";
    logEntries += "User " + userName + " registered<br>";
    nameEntryRequired = false;
    bookEntryRequired = true;
    userID++;
  } else {
    currentMessage = "Please enter a name.";
  }
}

// Handle book submission for borrowing or returning
void handleBookSubmission() {
  if (!bookEntryRequired) return;

  if (server.hasArg("book")) {
    String bookTitle = server.arg("book");
    bool found = false;

    for (int i = 0; i < MAX_BOOKS; i++) {
      if (books[i] == bookTitle) {
        found = true;
        if (bookAvailable[i]) {
          // Borrow the book
          bookAvailable[i] = false;
          currentMessage = bookTitle + " has been borrowed!";
          logEntries += bookTitle + " borrowed<br>";
        } else {
          // Return the book
          bookAvailable[i] = true;
          currentMessage = bookTitle + " has been returned!";
          logEntries += bookTitle + " returned<br>";
        }
        break;
      }
    }
    if (!found) {
      currentMessage = "Book title not recognized.";
    }

    bookEntryRequired = false; // Reset after borrowing or returning
  } else {
    currentMessage = "Please enter a book title.";
  }
}

// Serve the current status and logs as JSON for AJAX requests
void handleStatus() {
  String json = "{ \"message\": \"" + currentMessage + "\", \"bookList\": \"" + getAvailableBooks() + "\", \"logs\": \"" + logEntries + "\", \"nameEntryRequired\": " + String(nameEntryRequired) + ", \"bookEntryRequired\": " + String(bookEntryRequired) + " }";

}

void setup() {
  Serial.begin(115200);
  Serial2.begin(57600, SERIAL_8N1, RX_PIN, TX_PIN);
  finger.begin(57600);

  if (finger.verifyPassword()) {
    Serial.println("Fingerprint sensor ready.");
  } else {
    Serial.println("Fingerprint sensor not detected. Check connections.");
    while (1);
  }

  WiFi.softAPConfig(local_IP, gateway, subnet);
  WiFi.softAP(ssid, password);
  Serial.println("Access Point Started");
  Serial.println(WiFi.softAPIP());

  server.on("/", handleRoot);
  server.on("/status", handleStatus);
  server.on("/initiateRegister", initiateRegistration);
  server.on("/cancelRegister", cancelRegistration);
  server.on("/submitName", handleNameSubmission);
  server.on("/submitBook", handleBookSubmission);
  server.begin();
}

void loop() {
  server.handleClient();
  registerFingerprint();
  checkFingerprint();
}

   •setup()Configures the ESP32 as an Access Point, initializes the fingerprint sensor, and starts the web server to serve the interface.
   •loop()Continuously handles client requests, manages fingerprint registration steps, and checks for user identity to enable borrowing or returning.

ASSESSMENT

In this section, students are required to accomplish the following activity.

Project Description
The following assessment is based on the Library Management System project. It includes an intentional bug in the handleBookSubmission function, where the system fails to update the availability of the books correctly. Students must debug and fix the issue to ensure the system works as intended.

Scenario
You are tasked with maintaining and debugging a Library Management System. The system allows users to register with a fingerprint sensor and borrow or return books. Recently, the system was updated, but the book availability list no longer updates correctly.

You are tasked to:
   - Identify the bug in the provided code.
   - Fix the bug so that the book availability list on Adafruit IO updates properly when a book is borrowed or returned.

Code with Bug
The bug is in the handleBookSubmission function. Examine the code snippet below carefully.


void handleBookSubmission(String bookTitle) {
  bool found = false;
  for (int i = 0; i < MAX_BOOKS; i++) {
    if (books[i] == bookTitle) {
      found = true;
      if (bookAvailable[i]) {
        // Borrow the book
        bookAvailable[i] = true; // BUG: Incorrectly sets to 'true' instead of 'false'
        logEntries += bookTitle + " borrowed<br>";
        attendanceFeed->save(bookTitle + " borrowed.");
        currentMessage = bookTitle + " borrowed.";
      } else {
        // Return the book
        bookAvailable[i] = true;
        logEntries += bookTitle + " returned<br>";
        attendanceFeed->save(bookTitle + " returned.");
        currentMessage = bookTitle + " returned.";
      }
      break;
    }
  }

  if (!found) {
    currentMessage = "Book not found.";
  }

  updateAvailableBooks(); // Update the list of available books
  statusFeed->save(currentMessage);
}

Expected Fix

Replace the erroneous line in the handleBookSubmission function:
bookAvailable[i] = true; // BUG: Incorrectly sets to 'true' instead of 'false'
With:
bookAvailable[i] = false; // FIX: Correctly sets to 'false' when a book is borrowed
